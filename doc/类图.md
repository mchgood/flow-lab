%% =========================
%% Definition Layer
%% =========================

class ProcessDefinition {
    +String id
    +Map~String, Node~ nodes
    +Map~String, Edge~ edges
    +Map~String, List~Edge~~ outgoingIndex
    +Map~String, List~Edge~~ incomingIndex
}

class Node {
    +String id
    +NodeType type
    +GatewayType gatewayType
    +Map~String, Object~ metadata
}

class Edge {
    +String id
    +String sourceRef
    +String targetRef
    +String conditionExpression
    +boolean isDefault
}

ProcessDefinition --> Node
ProcessDefinition --> Edge


%% =========================
%% Runtime Layer
%% =========================

class ProcessInstance {
    +String instanceId
    +Map~String, Execution~ executions
    +Map~String, ScopeRuntime~ scopes
    +Map~String, Object~ variables
    +DataBus dataBus
}

class Execution {
    +String executionId
    +String currentNodeId
    +String scopeId
    +ExecutionStatus status
}

class ScopeRuntime {
    +String scopeId
    +ScopeStatus status
    +Duration timeout
}

ProcessInstance --> Execution
ProcessInstance --> ScopeRuntime

%% Task Layer (Spring)
%% =========================

class FlowTask {
    <<interface>>
    +execute(TaskContext)
}

class TaskContext {
    +ProcessInstance instance
    +Execution execution
    +Map~String,Object~ variables
    +DataBus dataBus
}

class DataBus {
    <<interface>>
    +put(key, value)
    +get(key)
    +remove(key)
}

class InMemoryDataBus
